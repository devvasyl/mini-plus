# mini-vue3
a mini vue3.0

```
npm init -y

npm install typescript --save-dev
yarn add typescript --dev

npx tsc --init

yarn add jest @types/jest --dev

// tsconfig.json
"types": ["jest"], 

// package.json
"scripts": {
    "test": "jest"
}

// tsconfig.json
 "noImplicitAny": false

 yarn add --dev babel-jest @babel/core @babel/preset-env

 yarn add --dev @babel/preset-typescript

```
